  <meta name="viewport" content="width=device-width, initial-scale=1.0">

  <title>Clear Personal AI Data</title>

</head>

<body>

  <p id="status">准备清理 Personal AI 本地数据…</p>


  <script>

    (async function clearPersonalAIDataOnce() {

      const status = document.getElementById('status');


      const confirmed = confirm(

        '确定清除这个浏览器中的 Personal AI 本地数据吗？\n\n' +

        '将删除：\n' +

        '- 聊天和消息\n' +

        '- 聊天顺序\n' +

        '- Anthropic API Key\n' +

        '- ElevenLabs Key\n' +

        '- 记忆设置内容\n' +

        '- 当前皮肤选择\n' +

        '- 本地语音缓存\n\n' +

        '不会修改 GitHub 上的 HTML 代码。'

      );


      if (!confirmed) {

        status.textContent = '已取消，没有删除任何数据。';

        return;

      }


      try {

        const keysToDelete = [

          'claude_key',

          'claude_memory',

          'claude_chats',

          'claude_order',

          'el_key',

          'claude_skin'

        ];


        keysToDelete.forEach(key => {

          localStorage.removeItem(key);

        });


        status.textContent = '文字数据已清除，正在删除语音缓存…';


        await new Promise((resolve, reject) => {

          const request = indexedDB.deleteDatabase('claude_audio');


          request.onsuccess = () => resolve();


          request.onerror = () => {

            reject(request.error || new Error('IndexedDB delete failed'));

          };


          request.onblocked = () => {

            reject(

              new Error(

                '语音数据库删除被其他仍打开的 Personal AI 页面阻塞。'

              )

            );

          };

        });


        status.textContent =

          '清理完成。请关闭所有 Personal AI 页面，然后重新打开正式入口。';


        alert(

          'Personal AI 本地数据已清除。\n\n' +

          '请关闭所有 Personal AI 标签页，再重新打开正式入口。'

        );


      } catch (error) {

        status.textContent =

          '文字数据已清除，但语音缓存删除可能未完成：' +

          String(error);


        alert(

          '聊天等 localStorage 数据已清除。\n\n' +

          '但语音缓存删除可能未完成。\n' +

          '请先关闭其他 Personal AI 标签页，再重新打开此清理页面重试。\n\n' +

          String(error)

        );

      }

    })();

  </script>

</body>

</html>
